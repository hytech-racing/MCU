173/* Send CAN messages */
174template<typename T>
175void TelemetryInterface::enqueue_CAN(T msg_class, uint32_t id) {
176
177 CAN_message_t msg;
178 msg_class.write(msg.buf);
179 msg.id = id;
180 msg.len = sizeof(msg_class);
181 uint8_t buf[sizeof(CAN_message_t)];
182 memmove(buf, &msg, sizeof(CAN_message_t));
183
184 msg_queue_->push_back(buf, sizeof(CAN_message_t));
185
186}

188/* Send inverter CAN messages with new CAN library */
189template<typename U>
190void TelemetryInterface::enqueue_new_CAN(U* structure, uint32_t (* pack_function)(U*, uint8_t*, uint8_t*, uint8_t*)) {
191 CAN_message_t can_msg;
192 can_msg.id = pack_function(structure, can_msg.buf, &can_msg.len, (uint8_t*) &can_msg.flags.extended);
193 uint8_t buf[sizeof(CAN_message_t)] = {};
194 memmove(buf, &can_msg, sizeof(CAN_message_t));
195 msg_queue_->push_back(buf, sizeof(CAN_message_t));
196}
