189template<typename T>
190void TelemetryInterface::enqueue_CAN(T msg_class, uint32_t id) {
191
192 CAN_message_t msg;
193 msg_class.write(msg.buf);
194 msg.id = id;
195 msg.len = sizeof(msg_class);
196 uint8_t buf[sizeof(CAN_message_t)];
197 memmove(buf, &msg, sizeof(CAN_message_t));
198
199 msg_queue_->push_back(buf, sizeof(CAN_message_t));
200
201}

204template<typename U>
205void TelemetryInterface::enqueue_new_CAN(U* structure, uint32_t (* pack_function)(U*, uint8_t*, uint8_t*, uint8_t*)) {
206 CAN_message_t can_msg;
207 can_msg.id = pack_function(structure, can_msg.buf, &can_msg.len, (uint8_t*) &can_msg.flags.extended);
208 uint8_t buf[sizeof(CAN_message_t)] = {};
209 memmove(buf, &can_msg, sizeof(CAN_message_t));
210 msg_queue_->push_back(buf, sizeof(CAN_message_t));
211}
