106TEST(SteeringSystemTesting, test_steering_divergence_warning)
107{
108 // Boilerplate definitions for the testing aparatus
109 SysClock sys_clock;
110 SysTick_s sys_tick = sys_clock.tick(0);
111 SteeringEncoderInterface primarySensor;
112 SteeringSystem steeringSystem(&primarySensor, PRIMARY_ALPHA, SECONDARY_ALPHA);
113 // Filter refernce values for evaluation
114 Filter_IIR<float> mockAngleFilters[2] = {PRIMARY_ALPHA, SECONDARY_ALPHA};
115 float mockPrimaryFilteredAngle;
116 float mockSecondaryFilteredAngle;
117
118 float primaryAngle = 0.0f;
119 // Pend divergence to sensor readings
120 // - IIR filter eq.: filtered_val = (1-alpha)*new_val + alpha*prev_val
121 // - when divergence set exactly to system threshold
122 // - divergence magnitude gets shaved off, will not trigger above threshold
123 // - pend offset to divergence to trip threshold
124 float secondaryAngle = primaryAngle + WARN_DIVERGENCE_OFFSET + STEERING_DIVERGENCE_WARN_THRESHOLD; // edge discrepancy
125
126 // Latency due to filtering
127 // - required steps of filtering for divergence to rise up to threshold with offset pended
128 // - e.g. when alpha = 0.7, pending offset = 0.6, it takes a minimal of 8 steps of filtering to exceed warn threshold
129 for (int i = 0; i < WARN_FILTER_LATENCY; i++) // edge filter steps to align output
130 {
131 mockPrimaryFilteredAngle = mockAngleFilters[0].filtered_result(primaryAngle);
132 mockSecondaryFilteredAngle = mockAngleFilters[1].filtered_result(secondaryAngle);
133
134 primarySensor.mockConversion.angle = primaryAngle;
135 primarySensor.mockConversion.status = SteeringEncoderStatus_e::STEERING_ENCODER_NOMINAL;
136 steeringSystem.tick(
137 (SteeringSystemTick_s)
138 {
139 .tick = sys_tick,
140 .secondaryConversion = (AnalogConversion_s)
141 {
142 .raw = 0,
143 .conversion = secondaryAngle,
144 .status = AnalogSensorStatus_e::ANALOG_SENSOR_GOOD
145 }
146 }
147 );
148 }
149
150 ASSERT_TRUE(steeringSystem.getSteeringSystemData().angle == mockPrimaryFilteredAngle);
151 ASSERT_TRUE(steeringSystem.getSteeringSystemData().status == SteeringSystemStatus_e::STEERING_SYSTEM_MARGINAL);
152}
