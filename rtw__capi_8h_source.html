170} rtwCAPI_DataTypeMap;

172/* Notes on rtwCAPI_DataTypeMap:
173 * cDataName - The ANSI C equivalent data type.
174 * For fixed point data
175 * cDataName = the native integer equivalent
176 * For Complex data or non-virtual structures,
177 * cDataName = "struct"
178 * mwDataName - RTW defined data type. Typedef can be found in rtwtypes.h
179 * For fixed point data,
180 * mwDataName = the integer type in which the data is stored.
181 * Use rtwCAPI_FixPtMap to get slope & bias of the data.
182 * numElements - Number of elements/members in the data type. If the data type
183 * represents a non-virtual structure (for e.g, bus structure),
184 * this field gives the number of elements in the structure.
185 * For non-structure data type, the default value is 0
186 * Complex data types are treated as non-structures
187 * elemMapIndex - Index into the rtwCAPI_ElementMap which maps each of the
188 * element/member of a structure data type.
189 * slDataId - Enumerated Simulink Data type defined in simstruc_types.h.
190 * The enumerations are
191 * o SS_DOUBLE
192 * o SS_SINGLE
193 * o SS_INT8
194 * o SS_UINT8
195 * o SS_INT16
196 * o SS_UINT16
197 * o SS_INT32
198 * o SS_UINT32
199 * o SS_BOOLEAN
200 * o SS_ENUM_TYPE (for enumerated data types)
201 * o SS_STRUCT (for non-virtual structures)
202 * For fixed point data:
203 * slDataId = enumeration corresponding to the integer type
204 * in which the data is stored
205 * For Complex data:
206 * slDataId = enumeration corresponding to the data type of
207 * the real part.
208 * isComplex - is the data type complex (1=Complex, 0=Real)
209 * isPointer - is data accessed Via Pointer (1=yes, 0= no)
210 */
