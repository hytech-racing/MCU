28/*
29 * System ran breadcrumb macros
30 *
31 * Clearing macro. 2 callers
32 * i) top of the sim-loop
33 * ii) top of each Outputs function for External mode
34 *
35 *
36 * Here is the basic idea:
37 * If a system is DISABLED, it stays DISABLED.
38 * For any other state, assume ENABLE_TO_DISABLE,
39 * If it runs then the OutputFcn will move it to ENABLE
40 * otherwise it will come back as ENABLE_TO_DISABLE(hence it never ran)
41 *
42 * DISABLE -> DISABLE (latched)
43 * ENABLE -> ENABLE_TO_DISABLE (assume)
44 * ENABLE_TO_DISABLE -> DISABLE (new latch)
45 * DISABLE_TO_ENABLE -> ENABLE (new enable)
46 * ONE_SHOT -> DISABLE (by definition)
47 *
48 * Note: These macros are called both by Simulink and the generated code
49 */
50
51/* sr stands for SystemRan, BC stands for BreadCrumb */
52#define srClearBC(state) \
53 { \
54 SubSystemRanBCTransition prevState = (SubSystemRanBCTransition)state; \
55 SubSystemRanBCTransition newState = SUBSYS_RAN_BC_DISABLE; \
56 \
57 switch (prevState) { \
58 case SUBSYS_RAN_BC_DISABLE: \
59 newState = SUBSYS_RAN_BC_DISABLE; \
60 break; \
61 case SUBSYS_RAN_BC_ENABLE: \
62 newState = SUBSYS_RAN_BC_ENABLE_TO_DISABLE; \
63 break; \
64 case SUBSYS_RAN_BC_DISABLE_TO_ENABLE: \
65 newState = SUBSYS_RAN_BC_ENABLE_TO_DISABLE; \
66 break; \
67 case SUBSYS_RAN_BC_ENABLE_TO_DISABLE: \
68 newState = SUBSYS_RAN_BC_DISABLE; \
69 break; \
70 case SUBSYS_RAN_BC_ONE_SHOT: \
71 newState = SUBSYS_RAN_BC_DISABLE; \
72 break; \
73 } \
74 state = newState; \
75 }

77/*
78 * Update macro

80 * Called by the OutputFcn of all conditionally exec'd subsystems
81 * from subsystm.cpp and commonbodlib.tlc

83 * DISABLE -> DISABLE_ENABLE (new enable)
84 * ENABLE_TO_DISABLE -> ENABLE (must have run last time step)
85 * All other state -> should really be an assert
86 *
87 */
