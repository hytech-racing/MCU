102void TorqueControllerMux::applyRegenLimit(DrivetrainCommand_s* command, const DrivetrainDynamicReport_s* drivetrain)
103{
104 const float noRegenLimitKPH = 10.0;
105 const float fullRegenLimitKPH = 5.0;
106 float maxWheelSpeed = 0.0;
107 float torqueScaleDown = 0.0;
108 bool allWheelsRegen = true; // true when all wheels are targeting speeds below the current wheel speed
109
110 for (int i = 0; i < NUM_MOTORS; i++) {
111 maxWheelSpeed = std::max(maxWheelSpeed, drivetrain->measuredSpeeds[i] * RPM_TO_KILOMETERS_PER_HOUR);
112 allWheelsRegen &= (command->speeds_rpm[i] < drivetrain->measuredSpeeds[i]);
113 }
114
115 // begin limiting regen at noRegenLimitKPH and completely limit regen at fullRegenLimitKPH
116 // linearly interpolate the scale factor between noRegenLimitKPH and fullRegenLimitKPH
117 torqueScaleDown = std::min(1.0f, std::max(0.0f, (maxWheelSpeed - fullRegenLimitKPH) / (noRegenLimitKPH - fullRegenLimitKPH)));
118
119 if (allWheelsRegen)
120 {
121 for (int i = 0; i < NUM_MOTORS; i++) {
122 command->torqueSetpoints[i] *= torqueScaleDown;
123 }
124 }
125}

127/*
128 Apply power limit such that the mechanical power of all wheels never
129 exceeds the preset mechanical power limit. Scales all wheels down to
130 preserve functionality of torque controllers
131*/
132void TorqueControllerMux::applyPowerLimit(DrivetrainCommand_s* command, const DrivetrainDynamicReport_s* drivetrain)
133{
134 float net_torque_mag = 0;
135 float net_power = 0;
136
137 // calculate current mechanical power
138 for (int i = 0; i < NUM_MOTORS; i++) {
139 // get the total magnitude of torque from all 4 wheels
140 #ifdef ARDUINO_TEENSY41 // screw arduino.h macros
141 net_torque_mag += abs(command->torqueSetpoints[i]);
142 net_power += abs(command->torqueSetpoints[i] * (drivetrain->measuredSpeeds[i] * RPM_TO_RAD_PER_SECOND));
143 #else
144 // sum up net torque
145 net_torque_mag += std::abs(command->torqueSetpoints[i]);
146 // calculate P = T*w for each wheel and sum together
147 net_power += std::abs(command->torqueSetpoints[i] * (drivetrain->measuredSpeeds[i] * RPM_TO_RAD_PER_SECOND));
148 #endif
149 }
150
151 // only evaluate power limit if current power exceeds it
152 if (net_power > (MAX_POWER_LIMIT)) {
153 for (int i = 0; i < NUM_MOTORS; i++) {
154 // calculate the percent of total torque requested per wheel
155 float torque_percent = command->torqueSetpoints[i] / net_torque_mag;
156 // based on the torque percent and max power limit, get the max power each wheel can use
157 float power_per_corner = (torque_percent * MAX_POWER_LIMIT);
158 // power / omega (motor rad/s) to get torque per wheel
159 command->torqueSetpoints[i] = power_per_corner / (drivetrain->measuredSpeeds[i] * RPM_TO_RAD_PER_SECOND);
160 }
161 }
162
163}

165/*
166 Apply limit such that the average wheel torque is never above the max
167 torque allowed in the current torque mode.
168 This will uniformally scale down all torques as to not affect the functionality
169 of non-symmetrical torque controllers.
170*/
171void TorqueControllerMux::applyTorqueLimit(DrivetrainCommand_s* command)
172{
173 float max_torque = getMaxTorque();
174 float avg_torque = 0;
175
176 // get the average torque accross all 4 wheels
177 for (int i = 0; i < NUM_MOTORS; i++) {
178 #ifdef ARDUINO_TEENSY41 // screw arduino.h macros
179 avg_torque += abs(command->torqueSetpoints[i]);
180 #else
181 avg_torque += std::abs(command->torqueSetpoints[i]);
182 #endif
183 }
184 avg_torque /= NUM_MOTORS;
185
186 // if this is greather than the torque limit, scale down
187 if (avg_torque > max_torque) {
188 // get the scale of avg torque above max torque
189 float scale = avg_torque / max_torque;
190 // divide by scale to lower avg below max torque
191 for (int i = 0; i < NUM_MOTORS; i++) {
192 command->torqueSetpoints[i] /= scale;
193 }
194 }
195
196}
