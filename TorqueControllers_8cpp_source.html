259void TorqueControllerSimpleLaunch::calc_launch_algo(const vector_nav* vn_data) {
260 /*
261 Stolen launch algo from HT07. This ramps up the speed target over time.
262 launch rate target is m/s^2 and is the target acceleration rate
263 secs_since_launch takes the milliseconds since launch started and converts to sec
264 This is then converted to RPM for a speed target
265 There is an initial speed target that is your iitial instant acceleration on the wheels
266 */
267 float secs_since_launch = (float)(current_millis_ - time_of_launch_) / 1000.0;
268 launch_speed_target_ = (int16_t)((float) secs_since_launch * launch_rate_target_ * METERS_PER_SECOND_TO_RPM);
269 launch_speed_target_ += init_speed_target_;
270 launch_speed_target_ = std::min((int)AMK_MAX_RPM, std::max(0, (int)launch_speed_target_));
271}

273void TorqueControllerSlipLaunch::calc_launch_algo(const vector_nav* vn_data) {
274 // accelerate at constant speed for a period of time to get body velocity up
275 // may want to make this the ht07 launch algo
276
277 // makes sure that the car launches at the target launch speed
278 launch_speed_target_ = std::max(launch_speed_target_, (float)DEFAULT_LAUNCH_SPEED_TARGET);
279
280 /*
281 New slip-ratio based launch algorithm by Luke Chen. The basic idea
282 is to always be pushing the car a certain 'slip_ratio_' faster than
283 the car is currently going, theoretically always keeping the car in slip
284 */
285 // m/s
286 float new_speed_target = (1 + slip_ratio_) * (vn_data->velocity_x);
287 // rpm
288 new_speed_target *= METERS_PER_SECOND_TO_RPM;
289 // makes sure the car target speed never goes lower than prev. target
290 // allows for the vn to 'spool' up and us to get reliable vx data
291 launch_speed_target_ = std::max(launch_speed_target_, new_speed_target);
292}

294void TorqueControllerLookupLaunch::calc_launch_algo(const vector_nav* vn_data) {
295
296 launch_speed_target_ = std::max((float)DEFAULT_LAUNCH_SPEED_TARGET, launch_speed_target_);
297
298 double dx = vn_data->ecef_coords[0] - initial_ecef_x_;
299 double dy = vn_data->ecef_coords[1] - initial_ecef_y_;
300 double dz = vn_data->ecef_coords[2] - initial_ecef_z_;
301
302 double distance = sqrt((dx*dx) + (dy*dy) + (dz*dz));
303
304 /*
305 Distance-lookup launch algorithm. Takes in the vel_dist_lookup
306 generated from Luke's matlab/symlink to set speed targets based
307 on distance travelled from the start point.
308 This can also and may be better to replace with an integration
309 of body velocity.
310 */
311
312 uint32_t idx = (uint32_t)(distance * 10); // multiply by 10 to be used as index for meters in steps of 1/10
313 idx = std::min(idx, (uint32_t)(sizeof(vel_dist_lookup) / sizeof(float)));
314 float mps_target = vel_dist_lookup[idx];
315
316 float new_speed_target = mps_target * METERS_PER_SECOND_TO_RPM;
317 launch_speed_target_ = std::max(launch_speed_target_, new_speed_target);
318
319}
