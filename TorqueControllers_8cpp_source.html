70 // Calculate torque commands at 100hz
71 if (tick.triggers.trigger100)
72 {
73 // Apply FIR filter to load cell data
74 loadCellForcesRaw_[0][FIRCircBufferHead] = flLoadCellData.conversion;
75 loadCellForcesRaw_[1][FIRCircBufferHead] = frLoadCellData.conversion;
76 loadCellForcesRaw_[2][FIRCircBufferHead] = rlLoadCellData.conversion;
77 loadCellForcesRaw_[3][FIRCircBufferHead] = rrLoadCellData.conversion;
78
79 for (int i = 0; i < 4; i++)
80 {
81 loadCellForcesFiltered_[i] = 0.0f;
82 for (int FIROffset = 0; FIROffset < numFIRTaps_; FIROffset++)
83 {
84 int index = (FIRCircBufferHead + FIROffset) % numFIRTaps_;
85 loadCellForcesFiltered_[i] += loadCellForcesRaw_[i][index] * FIRTaps_[FIROffset];
86 }
87 }
88 FIRCircBufferHead = (FIRCircBufferHead + 1) % numFIRTaps_;
89 if (FIRCircBufferHead == numFIRTaps_ - 1)
90 FIRSaturated_ = true;
91
92 // Do sanity checks on raw data
93 loadCellsErrorCounter_[0] = flLoadCellData.raw != 4095 && flLoadCellData.status != AnalogSensorStatus_e::ANALOG_SENSOR_CLAMPED ? 0 : loadCellsErrorCounter_[0] + 1;
94 loadCellsErrorCounter_[1] = frLoadCellData.raw != 4095 && frLoadCellData.status != AnalogSensorStatus_e::ANALOG_SENSOR_CLAMPED ? 0 : loadCellsErrorCounter_[1] + 1;
95 loadCellsErrorCounter_[2] = rlLoadCellData.raw != 4095 && rlLoadCellData.status != AnalogSensorStatus_e::ANALOG_SENSOR_CLAMPED ? 0 : loadCellsErrorCounter_[2] + 1;
96 loadCellsErrorCounter_[3] = rrLoadCellData.raw != 4095 && rrLoadCellData.status != AnalogSensorStatus_e::ANALOG_SENSOR_CLAMPED ? 0 : loadCellsErrorCounter_[3] + 1;
97 ready_ = FIRSaturated_ && loadCellsErrorCounter_[0] < errorCountThreshold_ && loadCellsErrorCounter_[1] < errorCountThreshold_ && loadCellsErrorCounter_[2] < errorCountThreshold_ && loadCellsErrorCounter_[3] < errorCountThreshold_;
98
99 writeout_.ready = ready_;
100
101 if (ready_)
102 {
103 // Calculate total normal force
104 float sumNormalForce = 0.0f;
105 for (int i = 0; i < 4; i++)
106 {
107 sumNormalForce += loadCellForcesFiltered_[i];
108 }
109
110 // Both pedals are not pressed and no implausibility has been detected
111 // accelRequest goes between 1.0 and -1.0
112 float accelRequest = pedalsData.accelPercent - pedalsData.regenPercent;
113 float torquePool;
