176void loop()
177{
178 // get latest tick from sys clock
179 SysTick_s curr_tick = sys_clock.tick(micros());
180
181 // process received CAN messages
182 process_ring_buffer(CAN2_rxBuffer, CAN_receive_interfaces, curr_tick.millis);
183 process_ring_buffer(CAN3_rxBuffer, CAN_receive_interfaces, curr_tick.millis);
184
185 // tick interfaces
186 tick_all_interfaces(curr_tick);
187
188 // tick systems
189 tick_all_systems(curr_tick);
190
191 // // inverter procedure before entering state machine
192 // // reset inverters
193 if (dashboard.inverterResetButtonPressed() && drivetrain.drivetrain_error_occured())
194 {
195 hal_println("resetting errored drivetrain");
196 drivetrain.reset_drivetrain();
197 }
198 // tick state machine
199 fsm.tick_state_machine(curr_tick.millis);
200
201 // tick safety system
202 safety_system.software_shutdown(curr_tick);
203
204 // send CAN
205 send_all_CAN_msgs(CAN2_txBuffer, &INV_CAN);
206 send_all_CAN_msgs(CAN3_txBuffer, &TELEM_CAN);
207}

238void tick_all_interfaces(const SysTick_s &current_system_tick)
239{
240
241 TriggerBits_s t = current_system_tick.triggers;
242 if (t.trigger10) // 10Hz
243 {
244 // Serial.println("before buzzer");
245 dashboard.tick10(&main_ecu, int(fsm.get_state()),
246 buzzer.buzzer_is_on(),
247 drivetrain.drivetrain_error_occured(),
248 torque_controller_mux.getTorqueLimit());
249
250 main_ecu.tick(static_cast<int>(fsm.get_state()),
251 drivetrain.drivetrain_error_occured(),
252 safety_system.get_software_is_ok(),
253 static_cast<int>(torque_controller_mux.getTorqueLimit()),
254 torque_controller_mux.getMaxTorque(),
255 buzzer.buzzer_is_on(),
256 pedals_system.getPedalsSystemData(),
257 ams_interface.pack_charge_is_critical(),
258 dashboard.launchControlButtonPressed());
259 }
260 if (t.trigger50) // 50Hz
261 {
262 steering1.sample();
263 PedalsSystemData_s data2 = pedals_system.getPedalsSystemDataCopy();
264 telem_interface.tick(a1.get(), a2.get(), a3.get(), steering1.convert(), &inv.fl, &inv.fr, &inv.rl, &inv.rr, data2.accelImplausible, data2.brakeImplausible, data2.accelPercent, data2.brakePercent);
265 }
266
267 if (t.trigger100) // 100Hz
268 {
269
270 a1.tick();
271 a2.tick();
272 a3.tick();
273 }
274 // // Untriggered
275 main_ecu.read_mcu_status(); // should be executed at the same rate as state machine
276 // DO NOT call in main_ecu.tick()
277}
