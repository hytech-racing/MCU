150void loop()
151{
152 // get latest tick from sys clock
153 SysTick_s curr_tick = sys_clock.tick(micros());
154
155 // process received CAN messages
156 process_ring_buffer(CAN2_rxBuffer, CAN_receive_interfaces, curr_tick.millis);
157 process_ring_buffer(CAN3_rxBuffer, CAN_receive_interfaces, curr_tick.millis);
158
159 // tick interfaces
160 tick_all_interfaces(curr_tick);
161
162 // tick systems
163 tick_all_systems(curr_tick);
164
165 // inverter procedure before entering state machine
166 // reset inverters
167 drivetrain_reset();
168
169 // tick state machine
170 fsm.tick_state_machine(curr_tick.millis);
171
172 // tick safety system
173 safety_system.software_shutdown(curr_tick);
174
175 // send CAN
176 send_all_CAN_msgs(CAN2_txBuffer, &INV_CAN);
177 send_all_CAN_msgs(CAN3_txBuffer, &TELEM_CAN);
178}

208
209void tick_all_interfaces(const SysTick_s &current_system_tick)
210{
211
212 TriggerBits_s t = current_system_tick.triggers;
213
214 if (t.trigger10) // 10Hz
215 {
216 dashboard.soundBuzzer(buzzer.buzzer_is_on());
217 dashboard.write();
218
219 main_ecu.tick(static_cast<int>(fsm.get_state()),
220 drivetrain.drivetrain_error_occured(),
221 safety_system.get_software_is_ok(),
222 static_cast<int>(torque_controller_mux.getTorqueLimit()),
223 torque_controller_mux.getMaxTorque(),
224 buzzer.buzzer_is_on(),
225 pedals_system.getPedalsSystemData(),
226 ams_interface.pack_charge_is_critical(),
227 dashboard.launchControlButtonPressed());
228 }
229 if (t.trigger50) // 50Hz
230 {
231 telem_interface.tick(ADC.a1.get(), ADC.a2.get(), ADC.a3.get(), steering1.convert());
232 }
233
234 if (t.trigger100) // 100Hz
235 {
236 ADC.a1.tick();
237 ADC.a2.tick();
238 ADC.a3.tick();
239 }
240
241 // Untriggered
242 main_ecu.read_mcu_status(); // should be executed at the same rate as state machine
243 // DO NOT call in main_ecu.tick()
244
245}
