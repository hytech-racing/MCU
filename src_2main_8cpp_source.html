302void loop()
303{
304 // get latest tick from sys clock
305 SysTick_s curr_tick = sys_clock.tick(micros());
306
307 handle_ethernet_interface_comms();
308
309 // process received CAN messages
310 process_ring_buffer(CAN2_rxBuffer, CAN_receive_interfaces, curr_tick.millis);
311 process_ring_buffer(CAN3_rxBuffer, CAN_receive_interfaces, curr_tick.millis);
312
313 // tick interfaces
314 tick_all_interfaces(curr_tick);
315
316 // tick systems
317 tick_all_systems(curr_tick);
318
319 // inverter procedure before entering state machine
320 // reset inverters
321 if (dashboard.inverterResetButtonPressed() && drivetrain.drivetrain_error_occured())
322 {
323 drivetrain.reset_drivetrain();
324 }
325 // tick state machine
326 fsm.tick_state_machine(curr_tick.millis);
327
328 // give the state of the car to the param interface
329 param_interface.update_car_state(fsm.get_state());
330
331 // tick safety system
332 safety_system.software_shutdown(curr_tick);
333
334 // send CAN
335 send_all_CAN_msgs(CAN2_txBuffer, &INV_CAN);
336 send_all_CAN_msgs(CAN3_txBuffer, &TELEM_CAN);
337}

368void tick_all_interfaces(const SysTick_s &current_system_tick)
369{
370
371 TriggerBits_s t = current_system_tick.triggers;
372 if (t.trigger10) // 10Hz
373 {
374 dashboard.tick10(
375 &main_ecu,
376 int(fsm.get_state()),
377 buzzer.buzzer_is_on(),
378 drivetrain.drivetrain_error_occured(),
379 torque_controller_mux.getTorqueLimit(),
380 ams_interface.get_filtered_min_cell_voltage(),
381 telem_interface.get_glv_voltage(a1.get()),
382 static_cast<int>(torque_controller_mux.activeController()->get_launch_state()),
383 dashboard.getDialMode());
384
385 main_ecu.tick(
386 static_cast<int>(fsm.get_state()),
387 drivetrain.drivetrain_error_occured(),
388 safety_system.get_software_is_ok(),
389 static_cast<int>(torque_controller_mux.getDriveMode()),
390 static_cast<int>(torque_controller_mux.getTorqueLimit()),
391 torque_controller_mux.getMaxTorque(),
392 buzzer.buzzer_is_on(),
393 pedals_system.getPedalsSystemData(),
394 ams_interface.pack_charge_is_critical(),
395 dashboard.launchControlButtonPressed());
396
397 PedalsSystemData_s data2 = pedals_system.getPedalsSystemDataCopy();
398
399 telem_interface.tick(
400 a1.get(),
401 a2.get(),
402 a3.get(),
403 steering1.convert(),
404 &inv.fl,
405 &inv.fr,
406 &inv.rl,
407 &inv.rr,
408 data2.accelImplausible,
409 data2.brakeImplausible,
410 data2.accelPercent,
411 data2.brakePercent,
412 a1.get().conversions[MCU15_ACCEL1_CHANNEL],
413 a1.get().conversions[MCU15_ACCEL2_CHANNEL],
414 a1.get().conversions[MCU15_BRAKE1_CHANNEL],
415 a1.get().conversions[MCU15_BRAKE2_CHANNEL],
416 pedals_system.getMechBrakeActiveThreshold(),
417 {});
418 }
419
420 if (t.trigger50) // 50Hz
421 {
422 steering1.sample();
423 }
424
425 if (t.trigger100) // 100Hz
426 {
427
428 a1.tick();
429 a2.tick();
430 a3.tick();
431 load_cell_interface.tick(
432 (LoadCellInterfaceTick_s){
433 .FLConversion = a2.get().conversions[MCU15_FL_LOADCELL_CHANNEL],

435 .RLConversion = sab_interface.rlLoadCell.convert(),
436 .RRConversion = sab_interface.rrLoadCell.convert()});
437 }
438 // // Untriggered
439 main_ecu.read_mcu_status(); // should be executed at the same rate as state machine
440 // DO NOT call in main_ecu.tick()
441}
