263 switch (mcu_status.get_state()) {
264 case MCU_STATE::STARTUP: break;
265
266 case MCU_STATE::TRACTIVE_SYSTEM_NOT_ACTIVE:
267#if DEBUG
268 Serial.println("TS NOT ACTIVE");
269#endif
270 // if TS is above HV threshold, move to Tractive System Active
271 if (check_TS_over_HV_threshold()) {
272#if DEBUG
273 Serial.println("Setting state to TS Active from TS Not Active");
274#endif
275 set_state(MCU_STATE::TRACTIVE_SYSTEM_ACTIVE);
276 }
277 break;
278
279 case MCU_STATE::TRACTIVE_SYSTEM_ACTIVE:
280 check_TS_active();
281 if (check_all_inverters_system_ready()) {
282 set_all_inverters_dc_on(true);
283 inverter_startup_state = INVERTER_STARTUP_STATE::WAIT_QUIT_DC_ON;
284 }
285 if (dashboard_status.get_start_btn() && mcu_status.get_mech_brake_active()) {
286#if DEBUG
287 Serial.println("Setting state to Enabling Inverter");
288#endif
289 set_state(MCU_STATE::ENABLING_INVERTER);
290 }
291 break;
292
293 case MCU_STATE::ENABLING_INVERTER:
294 check_TS_active();
295 // inverter enabling timed out
296 if (timer_inverter_enable.check()) {
297#if DEBUG
298 Serial.println("Setting state to TS Active from Enabling Inverter");
299#endif
300 set_state(MCU_STATE::TRACTIVE_SYSTEM_ACTIVE);
301 }
302
303 switch (inverter_startup_state) {
304 case INVERTER_STARTUP_STATE::WAIT_SYSTEM_READY:
305 if (check_all_inverters_system_ready()) {
306 set_all_inverters_dc_on(true);
307 inverter_startup_state = INVERTER_STARTUP_STATE::WAIT_QUIT_DC_ON;
308 }
309 break;
310
311 case INVERTER_STARTUP_STATE::WAIT_QUIT_DC_ON:
312 if (check_all_inverters_quit_dc_on()) {
313 set_all_inverters_no_torque();
314 set_all_inverters_driver_enable(true);
315 set_all_inverters_inverter_enable(true);
316 inverter_startup_state = INVERTER_STARTUP_STATE::WAIT_QUIT_INVERTER_ON;
317 }
318 break;
319
320 case INVERTER_STARTUP_STATE::WAIT_QUIT_INVERTER_ON:
321 if (check_all_inverters_quit_inverter_on()) {
322#if DEBUG
323 Serial.println("Setting state to Waiting Ready to Drive Sound");
324#endif
325 set_state(MCU_STATE::WAITING_READY_TO_DRIVE_SOUND);
326 }
327 break;
328
329 }
330 break;
331
332
333 case MCU_STATE::WAITING_READY_TO_DRIVE_SOUND:
334 check_TS_active();
335
336 // if the ready to drive sound has been playing for long enough, move to ready to drive mode
337 if (timer_ready_sound.check()) {
338#if DEBUG
339 Serial.println("Setting state to Ready to Drive");
340#endif
341 set_state(MCU_STATE::READY_TO_DRIVE);
342 }
343 break;
344
345 case MCU_STATE::READY_TO_DRIVE:
346 check_TS_active();
347

435void parse_telem_can_message(const CAN_message_t &RX_msg) {
436 CAN_message_t rx_msg = RX_msg;
437 switch (rx_msg.id) {
438 case ID_BMS_TEMPERATURES:
439 bms_temperatures.load(rx_msg.buf);
440 filtered_max_cell_temp = filtered_max_cell_temp * cell_temp_alpha + (1.0 - cell_temp_alpha) * (bms_temperatures.get_high_temperature() / 100.0) ;
441 break;
442 case ID_BMS_VOLTAGES:
443 bms_voltages.load(rx_msg.buf);
444 if (bms_voltages.get_low() < PACK_CHARGE_CRIT_LOWEST_CELL_THRESHOLD || bms_voltages.get_total() < PACK_CHARGE_CRIT_TOTAL_THRESHOLD) {
445 mcu_status.set_pack_charge_critical(true);
446 } else mcu_status.set_pack_charge_critical(false);
447 filtered_min_cell_voltage = filtered_min_cell_voltage * cell_voltage_alpha + (1.0 - cell_voltage_alpha) * (bms_voltages.get_low() / 10000.0);
448 break;
449 case ID_BMS_COULOMB_COUNTS: bms_coulomb_counts.load(rx_msg.buf); break;
450 case ID_BMS_STATUS:
451 bms_status.load(rx_msg.buf);
452 // BMS heartbeat timer
453 timer_bms_heartbeat.reset();
454 timer_bms_heartbeat.interval(BMS_HEARTBEAT_TIMEOUT);
455 break;
456 case ID_DASHBOARD_STATUS:
457 dashboard_status.load(rx_msg.buf);
458 /* process dashboard buttons */
459 if (dashboard_status.get_torque_mode_btn()) {
460 switch (mcu_status.get_torque_mode()) {
461 case 1:
462 mcu_status.set_max_torque(TORQUE_2);
463 mcu_status.set_torque_mode(2); break;
464 case 2:
465 mcu_status.set_max_torque(TORQUE_3);

467 case 3:
468 mcu_status.set_max_torque(TORQUE_1);

470 }
471 }

473 mcu_status.toggle_launch_ctrl_active();
474 }

476 inverter_restart = true;
477 timer_reset_inverter.reset();
478 }
479 // eliminate all action buttons to not process twice
480 dashboard_status.set_button_flags(0);
481 break;
482 }

1268inline void read_imu() {

1270 double sinAngle = sin(VEHICLE_TILT_ANGLE_X);
1271 double cosAngle = cos(VEHICLE_TILT_ANGLE_X);

1280 double vert_accel = -((accel_z * cosAngle) - (-accel_y * sinAngle));
1281 double pitch = (y_gyro * cosAngle) + (z_gyro * sinAngle);

1283 double yaw = (z_gyro * cosAngle) - (x_gyro * sinAngle);
